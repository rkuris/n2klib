182  unsigned long Get (int start, int len)
183  {
184  unsigned char startbit = start % 8;
185  unsigned char startbyte = start / 8;
186  // optimize for most common case, full bytes
187  if (startbit == 0)
188  {
189  switch (len)
190  {
191  case 8:
192  return data[startbyte];
193  case 16:
194  return intmem::loadu_le<uint16_t>(data + startbyte);
195  case 32:
196  return intmem::loadu_le<uint32_t>(data + startbyte);
197  case 64:
198  return intmem::loadu_le<uint64_t>(data + startbyte);
199  }
200  }
201  unsigned long result = 0;
202  int remain = len;
203  while (remain >= 8 && startbit == 0)
204  {
205  result = (result << 8) + data[startbyte];
206  startbyte++;
207  remain -= 8;
208  }
209  while (remain > 0)
210  {
211  result = result << 1;
212  result |= (data[startbyte] & (1 << startbit)) >> startbit;
213  startbit++;
214  if (startbit > 7)
215  {
216  startbit -= 8;
217  startbyte++;
218  }
219  remain--;
220  }
